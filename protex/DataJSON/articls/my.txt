        //GET
        public IActionResult Edit(int? id)
        {
            if (id == null || id == 0)
            {
                return NotFound();
            }

            var categoryFromBd = _db.Categories.Find(id);

            if (categoryFromBd == null)
            {
                return NotFound();
            }

            return View(categoryFromBd);
        }
